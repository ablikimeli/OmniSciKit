# OmniSciKit 使用案例

## 1. 环境管理

```r
library(OmniSciKit)

# 检查依赖包
check_deps(c("ggplot2", "dplyr", "survival"))

# 安装缺失的包
install_omni(c("pwr", "survival"))

# 设置选项
set_omni_options(verbose = TRUE)
get_omni_option("verbose")
```

## 2. 样本量计算

```r
# T检验样本量
ss_ttest(delta = 0.5, sd = 1, power = 0.8)
# 结果: 64

# ANOVA样本量
ss_anova(k = 3, f = 0.25, power = 0.8)
# 结果: 53

# 卡方检验样本量
ss_chisq(w = 0.3, df = 2, power = 0.8)
# 结果: 108

# 比例比较样本量
ss_proportion(p1 = 0.3, p2 = 0.5, power = 0.8)
# 结果: 93

# 相关性样本量
ss_correlation(r = 0.5, power = 0.8)
# 结果: 29

# 回归分析样本量
ss_regression(f2 = 0.15, u = 5, power = 0.8)
# 结果: 92

# 生存分析样本量
ss_survival(hr = 0.7, p = 0.5, power = 0.8)
# 结果: 247

# Logistic回归样本量
ss_logistic(p0 = 0.3, or = 1.5, power = 0.8)
# 结果: 425
```

## 3. 统计检验

```r
# T检验
set.seed(123)
x <- rnorm(30, mean = 5, sd = 2)
y <- rnorm(30, mean = 6, sd = 2)
stats_ttest(x, y)

# Wilcoxon检验
stats_wilcoxon(x, y)

# ANOVA
group <- rep(c("A", "B", "C"), each = 10)
value <- c(rnorm(10, 5), rnorm(10, 7), rnorm(10, 6))
stats_anova(value, group)

# 相关性分析
stats_correlation(rnorm(50), rnorm(50))

# 线性回归
x <- rnorm(50)
y <- 2*x + rnorm(50)
stats_regression(y, x)

# 卡方检验
tbl <- matrix(c(20, 30, 15, 35), nrow = 2)
stats_chisq(tbl)

# 正态性检验
stats_normality(rnorm(50))
```

## 5. 机器学习

```r
# 使用内置机器学习数据
data(omni_ml_data)
head(omni_ml_data)

# 数据分割
split <- ml_train_test_split(omni_ml_data, train_ratio = 0.8)
nrow(split$train)
nrow(split$test)

# 数据预处理
preproc <- ml_preprocess(omni_ml_data[, 1:5])
head(preproc$data)

# 训练模型
model <- ml_train(target ~ ., omni_ml_data[, 1:6], method = "rpart")
print(model)

# 预测
predictions <- ml_predict(model, omni_ml_data[1:10, ])
print(predictions)

# 评估
ml_evaluate(model, omni_ml_data[, 1:6])

# 交叉验证
ml_cv(target ~ ., omni_ml_data[, 1:6], method = "rpart", k = 5)
```

## 9. 综合案例：临床数据分析

```r
# 使用内置临床数据
data(omni_clinical)
head(omni_clinical)

# 1. 描述性统计
summary(omni_clinical[, c("age", "bmi", "systolic_bp")])

# 2. 组间比较
# T检验：不同治疗组的年龄
drug_a <- omni_clinical$age[omni_clinical$treatment == "Drug_A"]
drug_b <- omni_clinical$age[omni_clinical$treatment == "Drug_B"]
stats_ttest(drug_a, drug_b)

# 3. 生存分析
# 计算样本量
ss_survival(hr = 0.7, p = 0.5, power = 0.8)

# 4. 逻辑回归
# 预测治疗反应
model <- stats_logistic_reg(response ~ age + bmi + cholesterol, omni_clinical)
print(model)

# 5. 流行病学指标
# 计算应答率
responders <- sum(omni_clinical$response == "Responder")
total <- nrow(omni_clinical)
epi_attack_rate(responders, total)
```

## 10. 综合案例：疫情数据分析

```r
# 使用内置疫情数据

# 2. SIR模型拟合
sir_result <- epi_sir(
  N = omni_epidemic$population[1],
  I0 = omni_epidemic$new_cases[1],
  beta = 0.3,
  gamma = 0.1,
  days = max(omni_epidemic$day)
)

# 3. 预测
forecast <- epi_forecast(omni_epidemic$new_cases, days = 7)
print(forecast)

# 4. 疫苗效果
epi_vaccine_efficacy(50, 100, 80, 100)
```
